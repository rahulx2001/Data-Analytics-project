Bronze Layer Load

Bulk load CSV data into raw tables

SQL 
CALL bronze.load_bronze();

-- Internally executes:
COPY bronze.crm_cust_info 
FROM '/data/cust_info.csv'

3

Silver Layer Transform

Apply business rules and data quality transformations

SQL 
CALL silver.load_silver();

-- Transformations include:
SELECT 
    TRIM(cst_firstname) AS first_name,
    CASE 
        WHEN UPPER(cst_gndr) = 'M' THEN 'Male'
        WHEN UPPER(cst_gndr) = 'F' THEN 'Female'
        ELSE 'n/a'
    END AS gender
FROM bronze.crm_cust_info;
4

Gold Layer Model

Create star schema with dimensions and facts

SQL 
-- Create dimension views
CREATE VIEW gold.dim_customers AS
SELECT 
    ROW_NUMBER() OVER (ORDER BY cst_id) AS customer_key,
    ci.cst_id AS customer_id,
    ci.cst_key AS customer_number,
    ci.cst_firstname AS first_name,
    ci.cst_lastname AS last_name,
    ci.cst_marital_status AS marital_status,
    ci.cst_gndr AS gender,
    ca.bdate AS birth_date,
    la.cntry AS country
FROM silver.crm_cust_info ci
LEFT JOIN silver.erp_cust_az12 ca 
    ON ci.cst_key = ca.cid
LEFT JOIN silver.erp_loc_a101 la 
    ON ci.cst_key = la.cid;

Sample Quality Check Query

SQL 
-- Check for orphaned fact records (referential integrity)
SELECT 
    'Orphaned Records' AS check_name,
    COUNT(*) AS violations
FROM gold.fact_sales f
LEFT JOIN gold.dim_customers c ON c.customer_key = f.customer_key
LEFT JOIN gold.dim_products p ON p.product_key = f.product_key
WHERE p.product_key IS NULL 
   OR c.customer_key IS NULL;
